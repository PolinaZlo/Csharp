using System;

namespace laba_0_A
{
    class Program
    {
        private static int ReadInt(string prompt)
        {
            while (true)
            {
                Console.Write(prompt);
                int result;

                if (int.TryParse(Console.ReadLine(), out result))
                    return result;

                Console.ForegroundColor = ConsoleColor.Yellow; //уставновка цвета
                Console.WriteLine("\t¬веден неверный формат данных, повторите попытку!");
                Console.ResetColor(); //сбрасываем в стандартный
            }
        }

                Console.ResetColor(); //сбрасываем в стандартный
            }
        }
